# 聚合搜索

```
# 查看所有文档
GET user/_search
 
# 批量新增
PUT user/_bulk
{"index": {"_index": "user","_id": "1"}}
{"id": "1","name":"zq","age":20, "desc": "zq"}
{"index": {"_index": "user","_id": "2"}}
{"id": "2","name":"郑清","age": 18, "desc": "郑清"}
{"index": {"_index": "user","_id": "3"}}
{"id": "3","name":"zq努力学习","age":23, "desc": "zq 努力 学习"}
{"index": {"_index": "user","_id": "4"}}
{"id": "4","name":"天天向上","age": 18, "desc": "天天向上"}


# 查询 age>=18，再根据age分组聚合
GET user/_search
{
  "query": {
    "range": {
      "age": {
        "gte": 18
      }
    }
  },
  "aggs": {
    "aggAge": {
      "terms": {
        "field": "age"
      }
    }
  },
  "size": 0
}

# 根据age分组聚合，再对聚合后的结果按照age求平均值
GET user/_search
{
  "aggs": {
    "ageAgg": {
      "terms": {
        "field": "age"
      },
      "aggs": {
        "avgAgg": {
          "avg": {
            "field": "age"
          }
        }
      }
      
    }
  },
  "size": 0
}

# 获取结果集中的前N个数据
GET user/_search
{
  "aggs": {
    "top3": {
      "top_hits": {
        "size": 2
      }
    }
  },
  "size": 0
}

# 获取结果集中按照age字段排序后求取前N个数据
GET user/_search
{
  "aggs": {
    "top3": {
      "top_hits": {
        "sort": [
          {
          "age" : {
              "order":"desc"
            }
          }
        ], 
        "size": 3
      }
    }
  },
  "size": 0
}
```
